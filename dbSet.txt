CREATE SCHEMA `flyairplane` DEFAULT CHARACTER SET utf8 ;






CREATE TABLE `flyairplane`.`notice` (
  `idx` INT NOT NULL AUTO_INCREMENT DEFAULT 1,
  `title` VARCHAR(45) NOT NULL,
  `name` VARCHAR(45) NOT NULL,
  `content` VARCHAR(250) NULL,
  `wdate` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`idx`));

CREATE TABLE `flyairplane`.`admin` (
  `id` VARCHAR(45) NOT NULL,
  `password` VARCHAR(45) NOT NULL,
  `username` VARCHAR(45) NOT NULL,
  `nickname` VARCHAR(45) NOT NULL,
  PRIMARY KEY (`id`));
